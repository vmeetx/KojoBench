def draw_square_and_diagonals(sides: Int, length: Double) {
  var outer_turn = 360 / sides
  repeat(sides.toInt) {
    forward(length)
    right(outer_turn)
  }
  right(45)
  forward(length * math.sqrt(2))
  right(135)
  forward(length)
  right(135)
  forward(length * math.sqrt(2))
  right(45)
  forward(-length)
}
